  1. Panel plot comparing week days and weekend days

    Assignment conditions 8, 9

stats_for_steps_by_days_of_week <- data_tidy %>% 
  filter(steps > 0) %>% 
  group_by(date) %>% 
  summarize(avg_steps_per_day = mean(steps)) %>% 
  group_by(day_of_wk = as.factor(weekdays(date))) %>% 
  print()
